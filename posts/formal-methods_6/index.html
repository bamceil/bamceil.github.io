decideEquality() {
    for (each "x = y") {
        setx = find(x);
        sety = find(y);
        union(setx, sety);
    }
    while (stillChanging) {  // closure
        if ("s1 = t1", ..., "sn = tn") {
            union(f(s1, ..., sn), f(t1, ..., tn));
        }
    }
    for (each "x != y") {
        setx = find(x);
        sety = find(y);
        if(setx == sety)
            return UNSAT;
    }
    return SAT;
}
